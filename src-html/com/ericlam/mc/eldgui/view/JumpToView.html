001package com.ericlam.mc.eldgui.view;
002
003public final class JumpToView {
004
005    private final String view;
006    private final boolean keepPreviousUI;
007
008    public JumpToView(String view, boolean keepPreviousUI) {
009        this.view = view;
010        this.keepPreviousUI = keepPreviousUI;
011    }
012
013    public JumpToView(String view){
014        this(view, false);
015    }
016
017    public String getView() {
018        return view;
019    }
020
021    public boolean isKeepPreviousUI() {
022        return keepPreviousUI;
023    }
024}
